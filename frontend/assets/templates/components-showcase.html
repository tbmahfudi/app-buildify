How to Use Phosphor Icons

Basic Usage (Regular Weight)

<i class="ph ph-heart"></i>

Different Weights

<i class="ph-thin ph-heart"></i>     <!-- Thin weight -->
<i class="ph-light ph-heart"></i>    <!-- Light weight -->
<i class="ph ph-heart"></i>          <!-- Regular weight (default) -->
<i class="ph-bold ph-heart"></i>     <!-- Bold weight -->
<i class="ph-fill ph-heart"></i>     <!-- Fill weight -->
<i class="ph-duotone ph-heart"></i>  <!-- Duotone weight -->

Sizing with Tailwind

<i class="ph ph-star text-3xl"></i>    <!-- 1.875rem -->
<i class="ph ph-star text-4xl"></i>    <!-- 2.25rem -->

FlexStack Usage 

import { FlexStack } from './layout/flex-stack.js';

// Horizontal stack with buttons
const stack = new FlexStack('#container', {
  direction: 'horizontal',
  gap: 3,
  align: 'center',
  justify: 'start',
  items: [
    { content: button1 },
    { content: button2 },
    { content: button3 }
  ]
});

// Responsive stack
const responsiveStack = new FlexStack('#container', {
  direction: { xs: 'vertical', md: 'horizontal' },
  gap: { xs: 2, md: 4 },
  divider: { enabled: true }
});

FlexGrid Usage 

import { FlexGrid } from './layout/flex-grid.js';

// Responsive grid
const grid = new FlexGrid('#container', {
  columns: { xs: 1, sm: 2, md: 3, lg: 4 },
  gap: 6,
  items: [
    { id: 'card1', content: card1, span: 2 },
    { id: 'card2', content: card2, span: 1 },
    { id: 'card3', content: card3, span: 1 }
  ]
});

// Add item dynamically
grid.addItem({
  id: 'card4',
  content: card4,
  span: { xs: 1, md: 2 }
});

FlexContainer 

import FlexContainer from './layout/flex-container.js';

const container = new FlexContainer('#content', {
  preset: 'standard',  // narrow | standard | wide | full
  align: 'center',     // left | center | right
  gutter: true         // responsive padding
});

FlexSection 

import FlexSection from './layout/flex-section.js';

const section = new FlexSection('#hero', {
  variant: 'hero',     // hero | content | feature | cta
  padding: '2xl',      // sm | md | lg | xl | 2xl
  background: {
    type: 'gradient',
    gradient: { from: 'blue-600', to: 'indigo-600' }
  }
});

FlexSidebar 

import FlexSidebar from './layout/flex-sidebar.js';

const layout = new FlexSidebar('#app', {
  position: 'left',    // left | right | both
  width: { xs: '100%', md: '280px' },
  mobileMode: 'overlay',  // overlay | push
  desktopMode: 'push'     // push | overlay | reveal
});

Quick Reference

Basic usage examples for each component

FlexCard Usage 

import { FlexCard } from './components/flex-card.js';

const card = new FlexCard('#container', {
  title: 'Card Title',
  subtitle: 'Subtitle',
  icon: 'bi-star',
  badge: {text: 'New', variant: 'primary'},
  content: 'Card content here...',
  collapsible: true,
  actions: [
    {
      label: 'Edit',
      icon: 'bi-pencil',
      onClick: () => alert('Edit clicked')
    }
  ]
});

FlexModal Usage 

import { FlexModal } from './components/flex-modal.js';

const modal = new FlexModal({
  title: 'Modal Title',
  subtitle: 'Modal description',
  size: 'lg',
  content: 'Modal content...',
  footerActions: [
    {
      label: 'Cancel',
      variant: 'ghost',
      onClick: (e, modal) => modal.hide()
    },
    {
      label: 'Save',
      variant: 'primary',
      onClick: (e, modal) => {
        // Save logic
        modal.hide();
      }
    }
  ]
});

modal.show();

FlexTabs Usage 

import { FlexTabs } from './components/flex-tabs.js';

const tabs = new FlexTabs('#container', {

  tabs: [

      id: 'tab1',

    },
    {
      id: 'tab2',
      label: 'Tab Two',
      badge: {text: '5', variant: 'danger'},
      content: 'Tab 2 content...'
    }
  ]
});
